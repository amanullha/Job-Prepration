class Solution {
public:
    vector<int> twoSum(vector<int>& arr, int t) {
        
        
        unordered_map<int,int>seenBefore;
        
        for(int i=0;i<arr.size();i++){

            int x=t-arr[i];
            if(seenBefore.find(x)!=seenBefore.end() and seenBefore[x]!=i)
            {
                    return {i,seenBefore[x]};
            }else{
                
                seenBefore[arr[i]]=i;
            }
        }
        return {0,1};
        
    }
};
